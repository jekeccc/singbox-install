#!/bin/bash
# =============================================
# sing-box 多协议一键安装脚本（Grok 最终版）
# 支持 Hysteria2 端口跳跃 + WARP解锁 + 一键客户端二维码
# =============================================

set -e

# ================== 可自定义参数 ==================
DOMAIN=""                          # 填你的域名 或 留空使用IP
EMAIL="admin@example.com"
HY2_PASSWORD=$(openssl rand -hex 16)
HY2_PORT=443
HOP_PORT_START=20000
HOP_PORT_END=30000
MASQUERADE="https://www.bing.com"
# ================================================

RED='\033[0;31m'; GREEN='\033[0;32m'; YELLOW='\033[1;33m'; NC='\033[0m'
print() { echo -e "$1"; }

check_root() { [ "$(id -u)" -ne 0 ] && { print "${RED}请用 root 运行！${NC}"; exit 1; }; }

interactive_settings() {
    print "${GREEN}=== 端口跳跃设置 ===${NC}"
    read -p "主端口 [${HY2_PORT}]: " p; [ -n "$p" ] && HY2_PORT=$p
    read -p "跳跃起始端口 [${HOP_PORT_START}]: " s; [ -n "$s" ] && HOP_PORT_START=$s
    read -p "跳跃结束端口 [${HOP_PORT_END}]: " e; [ -n "$e" ] && HOP_PORT_END=$e
}

install_dependencies() {
    print "${GREEN}安装依赖（含 qrencode）...${NC}"
    apt update -y && apt install -y curl wget qrencode openssl jq uuid-runtime || \
    yum install -y curl wget qrencode openssl jq
}

setup_firewall_with_hop() {
    print "${GREEN}配置防火墙 + 端口跳跃...${NC}"
    iptables -I INPUT -p udp --dport $HY2_PORT -j ACCEPT
    iptables -I INPUT -p udp --dport $HOP_PORT_START:$HOP_PORT_END -j ACCEPT
    iptables -t nat -A PREROUTING -p udp --dport $HOP_PORT_START:$HOP_PORT_END -j REDIRECT --to-ports $HY2_PORT
    iptables-save > /etc/iptables/rules.v4
}

create_config() {
    print "${GREEN}生成服务端配置...${NC}"
    mkdir -p /etc/sing-box
    cat > /etc/sing-box/config.json << EOF
{
  "log": {"level": "info"},
  "inbounds": [{
    "type": "hysteria2",
    "tag": "hy2-in",
    "listen": "::",
    "listen_port": $HY2_PORT,
    "users": [{"password": "$HY2_PASSWORD"}],
    "masquerade": "$MASQUERADE",
    "tls": ${DOMAIN:+{"enabled":true,"acme":{"domains":["$DOMAIN"],"email":"$EMAIL"}}}
  }]
}
EOF
}

generate_client_qrcode() {
    print "${GREEN}🔄 正在生成客户端配置和二维码...${NC}"
    IP=$(curl -s4 ifconfig.me)
    SERVER=${DOMAIN:-$IP}

    cat > /root/hy2-client.yaml << EOF
server: $SERVER:$HY2_PORT,$HOP_PORT_START-$HOP_PORT_END

auth:
  type: password
  password: $HY2_PASSWORD

tls:
  sni: $SERVER
  insecure: false

transport:
  type: udp
  hopInterval: 30s
EOF

    # 生成二维码（终端显示 + PNG图片）
    qrencode -t ansiutf8 -l H < /root/hy2-client.yaml
    qrencode -t png -l H -o /root/hy2-client.png < /root/hy2-client.yaml

    print "${GREEN}✅ 客户端配置生成完成！${NC}"
    print "配置文件: /root/hy2-client.yaml"
    print "二维码图片: /root/hy2-client.png （可下载到手机扫码）"
}

# systemd 服务
setup_systemd() {
    cat > /etc/systemd/system/sing-box.service << EOF
[Unit]
Description=sing-box
After=network.target
[Service]
ExecStart=/usr/local/bin/sing-box run -c /etc/sing-box/config.json
Restart=always
[Install]
WantedBy=multi-user.target
EOF
    systemctl daemon-reload && systemctl enable --now sing-box
}

management_menu() {
    while true; do
        clear
        print "${GREEN}======== sing-box 管理菜单 =======${NC}"
        print "1. 查看服务状态"
        print "2. 重启服务"
        print "3. 生成客户端二维码（推荐）"
        print "4. 查看端口跳跃设置"
        print "5. 更新 sing-box"
        print "0. 退出"
        read -p "请选择: " choice
        case $choice in
            1) systemctl status sing-box ;;
            2) systemctl restart sing-box ;;
            3) generate_client_qrcode ;;
            4) print "主端口: $HY2_PORT | 跳跃: $HOP_PORT_START-$HOP_PORT_END" ;;
            5) bash <(curl -fsSL https://sing-box.app/install.sh) ;;
            0) exit 0 ;;
        esac
        read -p "按回车继续..."
    done
}

main() {
    check_root
    interactive_settings
    install_dependencies
    bash <(curl -fsSL https://sing-box.app/install.sh)
    setup_firewall_with_hop
    create_config
    setup_systemd
    generate_client_qrcode
    print "\n${GREEN}输入 ./$(basename "$0") menu 进入管理菜单${NC}"
}

if [ "$1" = "menu" ]; then
    management_menu
else
    main "$@"
fi
